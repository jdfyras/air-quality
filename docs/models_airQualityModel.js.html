const mongoose = require('mongoose');

/**
 * Mongoose Schema for air quality data.
 *
 * @typedef {Object} AirQualitySchema
 * @property {string} city - The name of the city.
 * @property {number} latitude - The latitude coordinates of the city.
 * @property {number} longitude - The longitude coordinates of the city.
 * @property {Date} timestamp - The timestamp of the air quality data.
 * @property {number} aqius - Air Quality Index (AQI) for the United States.
 * @property {string} mainus - Main pollutant for AQI in the United States.
 * @property {number} aqicn - Air Quality Index (AQI) for China.
 * @property {string} maincn - Main pollutant for AQI in China.
 */

const airQualitySchema = new mongoose.Schema({
    city: String,
    latitude: Number,
    longitude: Number,
    timestamp: Date,
    aqius: Number,
    mainus: String,
    aqicn: Number,
    maincn: String
});

airQualitySchema.set('toJSON', {
    virtuals: true
});

/**
 * Mongoose Model for air quality data.
 *
 * @type {import('mongoose').Model<AirQualitySchema>}
 */
const airQualityModel = mongoose.model('AirQuality', airQualitySchema);

module.exports = airQualityModel;
